    # Angular2中如何在发送delete请求时携带参数

    前端angular:

        /**
     * 根据ids删除资源分组
     */
    deleteResourceGroupByIds(ids: string[]) {
        return this.http.request('DELETE', this.API + '/resourceGroup', {
            body: ids
        }).pipe(
            retry(1),
            catchError(this.handleError.bind(this))
        );
    }
