    # img展示Blob类型数据

    component:

    export class XxxComponent {
    rqcode: any;

    getRqCode() {
        this.xxxService.getRqCode().subscribe(resp => {
            // blob转base64
            const reader = new FileReader();
            reader.readAsDataURL(resp);
            reader.onloadend = () => {
                this.rqcode = reader.result;
            };
        });
    }
}
